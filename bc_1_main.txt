                <input type="number" id="number1" placeholder="Enter first number" step="any">
                <div class="error-message" id="error1"></div>
            </div>
            
            <div class="input-group">
                <label for="number2">Second Number:</label>
                <input type="number" id="number2" placeholder="Enter second number" step="any">
                <div class="error-message" id="error2"></div>
            </div>
            
            <button type="button" id="calculateBtn">Calculate Sum</button>
        </form>

            <div class="result-label">Result:</div>
            <div class="result-value" id="result">0</div>

    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const number1Input = document.getElementById('number1');
            const number2Input = document.getElementById('number2');
            const error1Div = document.getElementById('error1');
            const error2Div = document.getElementById('error2');
            const calculateBtn = document.getElementById('calculateBtn');
            const resultDiv = document.getElementById('result');
            
            // Function to validate input
            function validateInput(inputElement, errorElement) {
                const value = inputElement.value.trim();
                
                // Clear previous error
                errorElement.textContent = '';
                inputElement.classList.remove('input-error');
                
                // Check if empty
                if (value === '') {
                    errorElement.textContent = 'Please enter a number';
                    inputElement.classList.add('input-error');
                    return false;
                }
                
                // Check if valid number
                if (isNaN(parseFloat(value))) {
                    errorElement.textContent = 'Please enter a valid number';
                    inputElement.classList.add('input-error');
                    return false;
                }
                
                return true;
            }
            
            // Function to add two numbers
            function addNumbers(num1, num2) {
                return parseFloat(num1) + parseFloat(num2);
            }
            
            // Function to update result display
            function updateResult(result) {
                resultDiv.textContent = result;
            }
            
            // Event listeners for input validation
            number1Input.addEventListener('blur', function() {
                validateInput(number1Input, error1Div);
            });
            
            number2Input.addEventListener('blur', function() {
                validateInput(number2Input, error2Div);
            });
            
            // Calculate button click event
            calculateBtn.addEventListener('click', function() {
                // Validate both inputs
                const isValid1 = validateInput(number1Input, error1Div);
                const isValid2 = validateInput(number2Input, error2Div);
                
                if (isValid1 && isValid2) {
                    // Perform addition
                    const sum = addNumbers(number1Input.value, number2Input.value);
                    
                    // Update result display
                    updateResult(sum);
                }
            });
            
            // Allow Enter key to trigger calculation
            document.addEventListener('keypress', function(e) {
                if (e.key === 'Enter') {
                    e.preventDefault();
                    calculateBtn.click();
                }
            });
        });
    </script>
